UDP Client-Server Python Code


import socket
import threading
import time

# UDP Server function
def udp_server():
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    server_socket.bind(('0.0.0.0', 12345))
    print("Server is running...")

    try:
        while True:
            data, client_address = server_socket.recvfrom(1024)
            message = data.decode('utf-8')
            print(f"[Server] Received: {message}")

            # Echo message back
            server_socket.sendto(data, client_address)
            print(f"[Server] Sent: {message}")

    except Exception as e:
        print("[Server] Exception:", e)

    finally:
        server_socket.close()

# Start server in a background thread
server_thread = threading.Thread(target=udp_server, daemon=True)
server_thread.start()

time.sleep(1)  # Give the server a moment to start

# UDP Client function
def udp_client():
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    server_host = '127.0.0.1'  # Localhost inside the Colab environment
    server_port = 12345

    try:
        while True:
            message = input("Enter message to send to server (or 'exit' to stop): ")
            if message.lower() == 'exit':
                break
            send_data = message.encode('utf-8')

            client_socket.sendto(send_data, (server_host, server_port))

            receive_data, _ = client_socket.recvfrom(1024)
            response = receive_data.decode('utf-8')
            print(f"[Client] Server says: {response}")

    except KeyboardInterrupt:
        print("\nClient stopped by user")

    finally:
        client_socket.close()

udp_client()
